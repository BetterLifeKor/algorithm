void displayV2(vector<vector<int>> v) {
	int n_r = v.size(); int n_c = v[0].size();
	for (int i = 0; i < n_r; i++) {
		for (int j = 0; j < n_c; j++) { printf("%d\t", v[i][j]); }
		printf("\n");
	}
}

void displayQ(queue<int> q) {
	queue<int> temp_q = q;
	while (!temp_q.empty()) {
		printf("%d\n", temp_q.front());
		temp_q.pop();
	}
}
