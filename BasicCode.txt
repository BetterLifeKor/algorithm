bool find_Q(queue<int> Q, int value) {
	bool flag = false;
	while (!Q.empty()) {
		if (Q.front() == value)
		{
			flag = true; break;
		}
		Q.pop();
	}
	return flag;
}
int find_Q2(queue<int> Q, queue<int> Q2, int value) {
	while (!Q.empty()) {
		if (Q.front() == value)	break;
		Q.pop(); Q2.pop();
	}
	return Q2.front();
}
int count_Q(queue<int> Q, int value) {
	int count = 0;
	while (!Q.empty()) {
		if (Q.front() == value) count++;
		Q.pop();
	}
	return count;
}
